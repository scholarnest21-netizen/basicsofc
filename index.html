Introduction to C

C is a general-purpose, procedural programming language developed by Dennis Ritchie at Bell Labs in 1972. It is the mother language of many modern languages — C++, Java, Python, and JavaScript all draw from C's design.

⚡

Fast & Efficient

Compiles directly to machine code with minimal overhead.

🔧

Low-Level Control

Direct memory management via pointers.

🌐

Portable

Write once, compile anywhere — ANSI C standard.

🏗️

Foundation

Linux kernel, databases, and OS kernels are written in C.

Program Structure

Every C program follows a defined structure. Understanding this skeleton is the first step to writing any program.

CAnatomy of a C Program
/* ① Header Files — include library functions */
#include <stdio.h>    // Standard I/O
#include <stdlib.h>   // Memory allocation, exit()
#include <string.h>   // String functions
#include <math.h>     // Mathematical functions

/* ② Global Declarations (visible to all functions) */
int globalVar = 10;

/* ③ Function Prototype (declaration before use) */
void greet(void);

/* ④ main() — Entry point of every C program */
int main() {
    // Local variable declaration
    int age = 20;

    // Statements and expressions
    greet();
    printf("Age: %d\n", age);

    return 0;    // 0 = success
}

/* ⑤ Function Definition */
void greet() {
    printf("Hello from ScholarNest!\n");
}
▶ Output
Hello from ScholarNest!
Age: 20

Variables & Constants

A variable is a named memory location that stores a value which can change. A constant is a value that never changes once defined.

Variable Declaration & Initialization

CVariables
// Declaration only
int num;

// Declaration + Initialization
int marks = 95;

// Multiple variables same line
int a = 5, b = 10, c;

// Variable naming rules:
// ✓ starts with letter or underscore
// ✓ can contain letters, digits, _
// ✗ cannot start with digit: 3num (invalid)
// ✗ cannot use keywords: int, float, etc.

int _count  = 0;     // valid
int myVar2  = 42;    // valid
// int 2var = 1;  ERROR — starts with digit

Constants — Three Ways

CConstants
// Method 1: #define (preprocessor macro)
#define PI       3.14159
#define MAX      100
#define GREETING "Hello!"

// Method 2: const keyword (type-safe, preferred)
const float GRAVITY = 9.8f;
const int   DAYS    = 365;

// Method 3: enum (named integer constants)
enum Weekday { MON=1, TUE, WED, THU, FRI, SAT, SUN };

int main() {
    printf("PI = %.5f\n", PI);
    printf("Gravity = %.1f m/s²\n", GRAVITY);

    enum Weekday today = WED;
    printf("Wednesday = %d\n", today); // prints 3

    return 0;
}
✅

Prefer const over #define for type safety. #define is a simple text replacement with no type checking.

Operators

C has a rich set of operators. They are the building blocks of expressions.

Arithmetic Operators

CArithmetic
int a = 10, b = 3;

printf("a + b  = %d\n", a + b);   // 13
printf("a - b  = %d\n", a - b);   // 7
printf("a * b  = %d\n", a * b);   // 30
printf("a / b  = %d\n", a / b);   // 3  (integer division)

int x = 5;
printf("%d\n", ++x);    // 6  (pre-increment)
printf("%d\n", x++);    // 6  (post-increment, prints then adds)
printf("%d\n", x);       // 7

Input & Output (I/O)

The stdio.h library provides the standard I/O functions. printf() writes to screen; scanf() reads from keyboard.

Cprintf() & scanf() — Complete Example
#include <stdio.h>

int main() {
    int   age;
    float gpa;
    char  name[50];

    // Reading a string (no & needed for char arrays)
    printf("Enter your name: ");
    scanf("%s", name);

    // Reading int and float
    printf("Enter age and GPA: ");
    scanf("%d %f", &age, &gpa);

    // Formatted output
    printf("\n--- Student Report ---\n");

    printf("Age  : %5d years\n", age); // right-align, 5 width
    printf("GPA  : %.2f\n", gpa);      // 2 decimal places

}

Conditional Statements

Conditionals let your program make decisions — executing different code depending on whether a condition is true or false.

if — else if — else Flowchart
Cif — else if — else Example
#include <stdio.h>

int main() {
    int marks;
    printf("Enter marks (0-100): ");
    scanf("%d", &marks);

    if (marks >= 90) {
        printf("Grade: A+ — Distinction!\n");
    } else if (marks >= 75) {
        printf("Grade: A — First Class\n");
    } else if (marks >= 60) {
        printf("Grade: B — Second Class\n");
    } else if (marks >= 40) {
        printf("Grade: C — Pass\n");
    } else {
        printf("Grade: F — Fail\n");
    }

    return 0;
}

switch — case Statement

Cswitch Example — Day of Week
#include <stdio.h>

int main() {
    int day;
    printf("Enter day (1-7): ");
    scanf("%d", &day);

    switch (day) {
        case 1: printf("Monday\n");    break;
        case 2: printf("Tuesday\n");   break;
        case 3: printf("Wednesday\n"); break;
        case 4: printf("Thursday\n");  break;
        case 5: printf("Friday\n");    break;
        case 6:
        case 7: printf("Weekend!\n");  break;
        default: printf("Invalid day\n");
    }
    return 0;
}
⚠️

Always use break in each case! Without it, execution "falls through" to the next case — this can cause bugs but is also intentionally used (cases 6 & 7 above share the same output).

Loops

Loops repeat a block of code. C provides three loop constructs: for, while, and do-while.

Three Loop Flowcharts — for / while / do-while
Cfor loop — Sum of N numbers
#include <stdio.h>

int main() {
    int n, i, sum = 0;
    printf("Enter n: ");
    scanf("%d", &n);

    for (i = 1; i <= n; i++) {
        sum += i;          // sum = sum + i
    }
    printf("Sum of 1 to %d = %d\n", n, sum);
    return 0;
}
▶ Output (n=5)
Sum of 1 to 5 = 15
Cwhile loop — Reverse a number
int n = 12345, rev = 0, rem;

while (n != 0) {
    rem = n % 10;      // extract last digit
    rev = rev * 10 + rem;
    n /= 10;            // remove last digit
}
printf("Reversed: %d\n", rev);   // 54321
Cdo-while — Menu until Exit
int choice;
do {
    printf("\n1. Add  2. Subtract  3. Exit\n");
    printf("Choice: ");
    scanf("%d", &choice);
    if (choice == 1)      printf("Adding...\n");
    else if (choice == 2) printf("Subtracting...\n");
} while (choice != 3);

printf("Goodbye!\n");

Arrays

An array is a collection of same-type elements stored in contiguous memory. Arrays are zero-indexed in C.

Array Memory Layout — int arr[5]
C1D Array — Declaration, Init, Traverse
#include <stdio.h>

int main() {
    // Declaration and initialization
    int arr[5] = {10, 20, 30, 40, 50};

    // Access elements
    printf("First: %d\n", arr[0]);    // 10
    printf("Last : %d\n", arr[4]);    // 50

    // Traverse with for loop
    int n = sizeof(arr) / sizeof(arr[0]);  // = 5
    printf("Array: ");
    for (int i = 0; i < n; i++)
        printf("%d ", arr[i]);

    // Find max value
    int max = arr[0];
    for (int i = 1; i < n; i++)
        if (arr[i] > max) max = arr[i];
    printf("\nMax = %d\n", max);

    return 0;
}
▶ Output
First: 10
Last : 50
Array: 10 20 30 40 50
Max = 50

Strings

In C, a string is a null-terminated character array. The last character is always '\0' (null character), which marks the end of the string.

CStrings — Declaration & Functions
#include <stdio.h>
#include <string.h>

int main() {
    char str1[20] = "Hello";
    char str2[20] = "World";
    char str3[50];

    printf("Length of str1: %zu\n", strlen(str1));  // 5

    strcpy(str3, str1);                    // copy str1 → str3
    printf("Copied: %s\n", str3);           // Hello

    strcat(str3, " ");
    strcat(str3, str2);                    // concatenate
    printf("Concatenated: %s\n", str3);    // Hello World

    int cmp = strcmp(str1, str2);          // compare
    if (cmp == 0) printf("Equal\n");
    else printf("Not equal (result=%d)\n", cmp);

    strupr(str1);   // HELLO (Windows/some compilers)
    strlwr(str2);   // world (Windows/some compilers)

    // Read string with spaces
    char fullname[50];
    printf("Enter full name: ");
    fgets(fullname, sizeof(fullname), stdin);  // safer than gets()

    return 0;
}

Functions

A function is a block of code that performs a specific task. Functions promote reusability, modularity, and readability.

Function Call & Return — Stack Frame
CFunctions — Complete Examples
#include <stdio.h>

// Function prototypes
int   add(int a, int b);
float average(int arr[], int n);
int   factorial(int n);         // recursive
void  swap(int *a, int *b);    // pointer param

int main() {
    printf("5+3 = %d\n", add(5, 3));

    int marks[] = {80, 75, 90, 65, 88};
    printf("Average = %.2f\n", average(marks, 5));

    printf("5! = %d\n", factorial(5));

    int x = 10, y = 20;
    swap(&x, &y);
    printf("x=%d y=%d\n", x, y);  // x=20 y=10

    return 0;
}

// Return sum
int add(int a, int b) {
    return a + b;
}

// Return average of array
float average(int arr[], int n) {
    int sum = 0;
    for (int i=0; i<n; i++) sum += arr[i];
    return (float)sum / n;
}

// Recursive factorial
int factorial(int n) {
    if (n <= 1) return 1;     // base case
    return n * factorial(n - 1); // recursive call
}

// Swap using pointers (pass by reference)
void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}
▶ Output
5+3 = 8
Average = 79.60
5! = 120

Pointers

A pointer is a variable that stores the memory address of another variable. Pointers are one of the most powerful — and tricky — features of C.

Pointer — Memory Address Diagram
CPointers — Core Concepts
#include <stdio.h>

int main() {
    int x = 42;
    int *p;        // declare pointer to int
    p = &x;        // p holds the address of x

    printf("Value of x   : %d\n", x);
    printf("Address of x : %p\n", (void*)&x);
    printf("p holds      : %p\n", (void*)p);
    printf("*p (deref)   : %d\n", *p);   // 42

    *p = 100;   // modify x through pointer
    printf("x is now     : %d\n", x);   // 100!

    // Pointer arithmetic
    int arr[] = {10, 20, 30};
    int *ptr = arr;       // arr decays to &arr[0]
    printf("arr[0]=%d, arr[1]=%d, arr[2]=%d\n",
           *ptr, *(ptr+1), *(ptr+2));

    return 0;
}
▶ Output
Value of x   : 42
Address of x : 0x7fff5fbff8ac
p holds      : 0x7fff5fbff8ac
*p (deref)   : 42
x is now     : 100
arr[0]=10, arr[1]=20, arr[2]=30
🚫

Never use an uninitialized pointer! Always assign a pointer before dereferencing it. Use NULL to mark an unused pointer: int *p = NULL; and check if (p != NULL) before use.

Structures

A struct groups related data of different types under one name — the foundation of user-defined data types in C.

CStructures — Student Record
#include <stdio.h>
#include <string.h>

// Define structure (blueprint)
struct Student {
    int   rollno;
    char  name[50];
    float marks;
    char  grade;
};

// typedef — avoids writing "struct" every time
typedef struct Student Student;

void printStudent(Student s);   // prototype

int main() {
    // Create and initialize struct variable
    Student s1 = {101, "Priya Sharma", 92.5f, 'A'};

    // Access members with dot operator
    printf("Roll No: %d\n", s1.rollno);
    printf("Name   : %s\n", s1.name);
    printf("Marks  : %.1f\n", s1.marks);

    // Modify a field
    s1.marks = 95.0f;

    // Array of structures
    Student class_a[3] = {
        {101, "Alice", 88.0, 'A'},
        {102, "Bob",   72.5, 'B'},
        {103, "Carol", 95.0, 'A'}
    };

    printf("\n--- Class Report ---\n");
    for (int i=0; i<3; i++)
        printStudent(class_a[i]);

    // Pointer to struct — use arrow operator
    Student *sp = &s1;
    printf("\nVia pointer: %s — %.1f\n", sp->name, sp->marks);

    return 0;
}

void printStudent(Student s) {
    printf("[%d] %-12s  %.1f  %c\n",
           s.rollno, s.name, s.marks, s.grade);
}
▶ Output
Roll No: 101
Name   : Priya Sharma
Marks  : 92.5

--- Class Report ---
[101] Alice        88.0  A
[102] Bob          72.5  B
[103] Carol        95.0  A

Via pointer: Priya Sharma — 95.0
📌

Use dot operator (.) to access struct members directly. Use arrow operator (->) when accessing via a pointer: ptr->member is shorthand for (*ptr).member.

File Input / Output

C can read from and write to files using FILE pointers from <stdio.h>. Always close files after use!

File I/O Workflow
CWriting to & Reading from a File
#include <stdio.h>
#include <stdlib.h>

int main() {
    FILE *fp;

    // ── WRITE to file ──
    fp = fopen("students.txt", "w");
    if (fp == NULL) {
        printf("Error opening file!\n");
        exit(1);
    }
    fprintf(fp, "Name: Alice\nMarks: 92.5\n");
    fprintf(fp, "Name: Bob\nMarks: 78.0\n");
    fclose(fp);
    printf("Data written to students.txt\n");

    // ── READ from file ──
    fp = fopen("students.txt", "r");
    if (fp == NULL) {
        printf("Cannot open file!\n");
        exit(1);
    }

    char line[100];
    printf("\n--- Reading File ---\n");
    while (fgets(line, sizeof(line), fp) != NULL) {
        printf("%s", line);
    }
    fclose(fp);

    return 0;
}
